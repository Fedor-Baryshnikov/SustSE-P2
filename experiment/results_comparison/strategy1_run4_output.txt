```python
from typing import Optional

from fastapi import FastAPI, HTTPException, status
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel, Field
from sqlalchemy import Column, Integer, String, Boolean, DateTime
from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy.orm import sessionmaker, Session
import datetime

app = FastAPI()

# CORS configuration (for local development)
origins = ["http://localhost:8000", "http://localhost:3000"]  # Adjust as needed
app.add_middleware(
    CORSMiddleware,
    allow_origins=origins,
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# SQLAlchemy setup
Base = declarative_base()

class Task(Base):
    __tablename__ = "tasks"

    id = Column(Integer, primary_key=True, index=True)
    title = Column(String, nullable=False)
    description = Column(String, default="")
    completed = Column(Boolean, default=False)
    created_at = Column(DateTime, default=datetime.datetime.now)

    def to_dict(self):
        return {
            "id": self.id,
            "title": self.title,
            "description": self.description,
            "completed": self.completed,
            "created_at": self.created_at,
        }

class TaskCreate(BaseModel):
    title: str = Field(..., min_length=1)
    description: Optional[str] = None

class TaskUpdate(BaseModel):
    title: Optional[str] = None
    description: Optional[str] = None
    completed: Optional[bool] = None

class TaskResponse(BaseModel):
    id: int
    title: str
    description: Optional[str] = None
    completed: bool
    created_at: datetime.datetime


# In-memory database (replace with actual database connection in production)
class DatabaseSession:
    def __init__(self):
        self.engine = None  #  Replace with your database URL (e.g., "sqlite:///./task_management.db")
        self.session = None

    def connect(self):
        # Replace with your actual database engine initialization
        # Example for SQLite:
        self.engine =  sqlite3.create_engine("sqlite:///./task_management.db")  # Replace this line
        Base.metadata.create_all(self.engine)
        self.session = sessionmaker(bind=self.engine)()

    def close(self):
        if self.session:
            self.session.close()

db_session = DatabaseSession()
db_session.connect()
Session = db_session.session


# CRUD operations
def create_task(title: str, description: Optional[str] = None) -> TaskResponse:
    task = Task(title=title, description=description)
    db_session.add(task)
    db_session.commit()
    db_session.refresh(task)
    return TaskResponse(**task.to_dict())


def read_task(task_id: int) -> Optional[TaskResponse]:
    task = db_session.query(Task).filter(Task.id == task_id).first()
    if task:
        return TaskResponse(**task.to_dict())
    return None


def update_task(task_id: int, update_data: TaskUpdate) -> TaskResponse:
    task = db_session.query(Task).filter(Task.id == task_id).first()
    if not task:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Task not found")

    for key, value in update_data.dict(exclude_unset=True).items():
        setattr(task, key, value)

    db_session.commit()
    db_session.refresh(task)
    return TaskResponse(**task.to_dict())


def delete_task(task_id: int):
    task = db_session.query(Task).filter(Task.id == task_id).first()
    if not task:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Task not found")

    db_session.delete(task)
    db_session.commit()


# API endpoints
@app.post("/tasks", response_model=TaskResponse, status_code=status.HTTP_201_CREATED)
async def create(task_data: TaskCreate):
    return create_task(title=task_data.title, description=task_data.description)


@app.get("/tasks/{task_id}", response_model=TaskResponse)
async def read(task_id: int):
    task = read_task(task_id)
    if task is None:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Task not found")
    return task


@app.put("/tasks/{task_id}", response_model=TaskResponse)
async def update(task_id: int, task_data: TaskUpdate):
    return update_task(task_id, task_data)


@app.delete("/tasks/{task_id}", status_code=status.HTTP_204_NO_CONTENT)
async def delete(task_id: int):
    delete_task(task_id)
    return


@app.get("/tasks", response_model=list[TaskResponse])
async def list_tasks():
    tasks = db_session.query(Task).all()
    return [TaskResponse(**task.to_dict()) for task in tasks]


if __name__ == "__main__":
    import sqlite3
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
```
