```python
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from sqlalchemy import Column, Integer, String, Boolean, DateTime
from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy.orm import sessionmaker, Session
from datetime import datetime
from typing import List
from . import models

Base = declarative_base()

class Task(models.Task):
    id: int = Column(Integer, primary_key, autoincrement=True)
    title: str
    description: str
    completed: bool
    created_at: DateTime = Column(DateTime, default=datetime.datetime.utcnow())

class TaskCreate(BaseModel):
    title: str
    description: str
    completed: bool = False

class TaskUpdate(BaseModel):
    title: str = None
    description: str = None
    completed: bool = None

class TaskResponse(BaseModel):
    id: int
    title: str
    description: str
    completed: bool
    created_at: datetime

app = FastAPI()

engine = models.Engine()
SessionLocal = sessionmaker(bind=engine)

db = Session()

@app.on_event("shutdown")
async def shutdown_event():
    db.close()

@app.get("/tasks", response_model=List[TaskResponse])
async def get_tasks():
    tasks = db.query(Task).all()
    return [TaskResponse(**task.__dict__) for task in tasks]

@app.get("/tasks/{task_id}", response_model=TaskResponse)
async def get_task(task_id: int):
    task = db.query(Task).filter(Task.id == task_id).first()
    if not task:
        raise HTTPException(status_code=404, detail="Task not found")
    return TaskResponse(**task.__dict__)

@app.post("/tasks", response_model=TaskResponse)
async def create_task(task: TaskCreate):
    new_task = Task(**task.dict())
    db.add(new_task)
    db.commit()
    return TaskResponse(**new_task.__dict__)

@app.put("/tasks/{task_id}", response_model=TaskResponse)
async def update_task(task_id: int, task: TaskUpdate):
    task_obj = db.query(Task).filter(Task.id == task_id).first()
    if not task_obj:
        raise HTTPException(status_code=404, detail="Task not found")
    
    if task.title is not None:
        task_obj.title = task.title
    if task.description is not None:
        task_obj.description = task.description
    if task.completed is not None:
        task_obj.completed = task.completed
    
    db.commit()
    return TaskResponse(**task_obj.__dict__)

@app.delete("/tasks/{task_id}", response_model=TaskResponse)
async def delete_task(task_id: int):
    task_obj = db.query(Task).filter(Task.id == task_id).first()
    if not task_obj:
        raise HTTPException(status_code=404, detail="Task not found")
    db.delete(task_obj)
    db.commit()
    return TaskResponse(id=task_obj.id, title=task_obj.title, description=task_obj.description, completed=task_obj.completed, created_at=task_obj.created_at)
```
