```python
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy import Column, Integer, String, Boolean, DateTime
from sqlalchemy.engine import create_engine
from datetime import datetime
from typing import List, Optional

# --- Database Setup ---
Base = declarative_base()

class Task(Base):
    __tablename__ = "tasks"
    id = Column(Integer, primary_key=True)
    title = Column(String)
    description = Column(String)
    completed = Column(Boolean, default=False)
    created_at = Column(DateTime, default=datetime.datetime.now)

    def __repr__(self):
        return f"<Task(id={self.id}, title='{self.title}', completed={self.completed})>"


def create_db_engine():
    engine = create_engine('sqlite:///./task_manager.db')
    Base.metadata.create_all(engine)
    return engine

# --- Pydantic Models ---

class TaskCreate(BaseModel):
    title: str
    description: str
    completed: bool = False

class TaskUpdate(BaseModel):
    title: Optional[str] = None
    description: Optional[str] = None
    completed: Optional[bool] = None

class TaskResponse(BaseModel):
    id: int
    title: str
    description: str
    completed: bool
    created_at: datetime


# --- API Implementation ---

app = FastAPI()
engine = create_db_engine()

@app.on_event("startup")
def startup_event():
    # Create the tasks table if it doesn't exist
    Base.metadata.create_all(engine)

@app.post("/tasks/", response_model=TaskResponse)
async def create_task(task: TaskCreate):
    new_task = Task(title=task.title, description=task.description, completed=task.completed)
    db = engine.connect()
    db.add(new_task)
    db.commit()
    db.close()
    return TaskResponse(id=new_task.id, title=new_task.title, description=new_task.description, completed=new_task.completed, created_at=new_task.created_at)


@app.get("/tasks/", response_model=List[TaskResponse])
async def get_tasks():
    tasks = []
    db = engine.connect()
    for task in db.query(Task).all():
        tasks.append(TaskResponse(id=task.id, title=task.title, description=task.description, completed=task.completed, created_at=task.created_at))
    db.close()
    return tasks


@app.get("/tasks/{task_id}", response_model=TaskResponse)
async def get_task(task_id: int):
    db = engine.connect()
    task = db.query(Task).filter_by(id=task_id).first()
    db.close()

    if task is None:
        raise HTTPException(status_code=404, detail="Task not found")

    return TaskResponse(id=task.id, title=task.title, description=task.description, completed=task.completed, created_at=task.created_at)


@app.put("/tasks/{task_id}", response_model=TaskResponse)
async def update_task(task_id: int, task_update: TaskUpdate):
    db = engine.connect()
    task = db.query(Task).filter_by(id=task_id).first()
    if task is None:
        raise HTTPException(status_code=404, detail="Task not found")

    if task_update.title is not None:
        task.title = task_update.title
    if task_update.description is not None:
        task.description = task_update.description
    if task_update.completed is not None:
        task.completed = task_update.completed

    db.commit()
    db.close()
    return TaskResponse(id=task.id, title=task.title, description=task.description, completed=task.completed, created_at=task.created_at)


@app.delete("/tasks/{task_id}", status_code=204)
async def delete_task(task_id: int):
    db = engine.connect()
    db.query(Task).filter_by(id=task_id).delete()
    db.commit()
    db.close()

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
```
