```python
from fastapi import FastAPI, HTTPException, status
from fastapi.middleware import CORS
from pydantic import BaseModel
from sqlalchemy import create_engine, Column, Integer, String, Boolean, DateTime
from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy.orm import sessionmaker
from datetime import datetime

app = FastAPI()
app.add_middleware(CORS)  # Enable CORS for all origins (for development)

# Database setup
DATABASE_URL = "sqlite:///./task_management.db"  # SQLite database file
engine = create_engine(DATABASE_URL)
Base = declarative_base()


class Task(Base):
    __tablename__ = "tasks"

    id = Column(Integer, primary_key=True, index=True)
    title = Column(String(255), index=True)
    description = Column(String(500), nullable=True)
    completed = Column(Boolean, default=False)
    created_at = Column(DateTime, default=datetime.now)

    def to_dict(self):
        return {
            "id": self.id,
            "title": self.title,
            "description": self.description,
            "completed": self.completed,
            "created_at": self.created_at,
        }


Base.metadata.create_all(engine)
Session = sessionmaker(bind=engine)
db = Session(bind=engine)


class TaskCreate(BaseModel):
    title: str
    description: str = None  # Optional description


class TaskUpdate(BaseModel):
    title: str = None
    description: str = None
    completed: bool = None


class TaskResponse(BaseModel):
    id: int
    title: str
    description: str
    completed: bool
    created_at: datetime


# --- API Endpoints ---

@app.post("/tasks", response_model=TaskResponse, status_code=status.HTTP_201_CREATED)
async def create_task(task_data: TaskCreate):
    """
    Creates a new task.
    """
    task = Task(title=task_data.title, description=task_data.description)
    db.add(task)
    db.commit()
    db.refresh(task)  # Refresh to get the assigned ID
    return task.to_dict()


@app.get("/tasks", response_model=list[TaskResponse])
async def read_tasks():
    """
    Retrieves all tasks.
    """
    tasks = db.query(Task).all()
    return [task.to_dict() for task in tasks]


@app.get("/tasks/{task_id}", response_model=TaskResponse)
async def read_task(task_id: int):
    """
    Retrieves a specific task by ID.
    """
    task = db.query(Task).filter(Task.id == task_id).first()
    if task is None:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Task not found")
    return task.to_dict()


@app.put("/tasks/{task_id}", response_model=TaskResponse)
async def update_task(task_id: int, task_data: TaskUpdate):
    """
    Updates an existing task.
    """
    task = db.query(Task).filter(Task.id == task_id).first()
    if task is None:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Task not found")

    if task_data.title:
        task.title = task_data.title
    if task_data.description:
        task.description = task_data.description
    if task_data.completed is not None:
        task.completed = task_data.completed

    db.commit()
    db.refresh(task)
    return task.to_dict()


@app.delete("/tasks/{task_id}", status_code=status.HTTP_204_NO_CONTENT)
async def delete_task(task_id: int):
    """
    Deletes a task.
    """
    task = db.query(Task).filter(Task.id == task_id).first()
    if task is None:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Task not found")
    db.delete(task)
    db.commit()
    return  # 204 No Content



if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
```
